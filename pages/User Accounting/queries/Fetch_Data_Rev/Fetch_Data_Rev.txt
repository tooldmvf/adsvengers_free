SELECT b.network_name,a.fk_network_id,a.fk_user_id,a.financial_month,amount_revenue_approved, a.commissions_fee
FROM public.user_financial_month a
inner join public.network b on (b.network_id = a.fk_network_id)
where 1 = 1
and b.network_name_short != 'AW'
and ({{!SelectUser.selectedOptionValue}} OR (text(a.fk_user_id) IN ({{"'" + SelectUser.selectedOptionValue + "'"}})))
AND SUBSTRING(a.financial_month,1,4) = '{{SelectYear.selectedOptionValue}}'
AND SUBSTRING(a.financial_month,5,2) = '{{SelectMonth.selectedOptionValue}}'
order by 1 asc
