select
  c.fk_bm_id,
  b.fk_ad_account_id,
  a.campaign_id,
  a.fk_user_id,
  a.campaign_offer_id,
  a.campaign_name,
  d.financial_date,
  a.spent_rate_fees,
  a.off_cat_short,
  a.net_name_short,
  g.off_cat_id
from
  public.raw_user_fb_campaign a
  inner join public.user_fb_account b on (
    a.fk_user_id = b.fk_user_id
    and a.fk_ad_account_id = b.fk_ad_account_id
  )
  inner join public.fb_ad_account c on (b.fk_ad_account_id = ad_account_id)
  inner join public.user_financial d on (
    a.fk_user_id = d.fk_user_id
    and a.campaign_offer_id = d.fk_offer_id
  )
  inner join public.user_network e on (
    a.fk_user_id = e.fk_user_id
    and e.user_network_id = d.fk_user_network_id
  )
  inner join public."network" f on (
    e.fk_network_id = f.network_id
    and f.network_name_short = a.net_name_short
  )
  inner join public.offer_category g on (g.off_cat_short = a.off_cat_short)
where
  d.financial_date >= '{{DatePicker2.selectedDate}}'
order by d.financial_date
