SELECT 'Revenue', a.fk_user_id, a.financial_month, sum(amount_revenue_approved) as appr_rev
FROM public.user_financial_month a
inner join public.network b on (b.network_id = a.fk_network_id)
where 1 = 1
and b.network_name_short != 'AW'
GROUP By a.fk_user_id, a.financial_month
UNION ALL
SELECT 'Spent', a.fk_user_id, a.financial_month, SUM(amount_spent)*(1+(a.spent_rate_fees/100::float))
FROM public.user_spent_month a
inner join public.facebook_bm b on (b.bm_id = a.fk_bm_id)
where 1 = 1
and fk_bm_id != 'Ai20-Google'
group by a.fk_user_id, a.financial_month,a.spent_rate_fees
